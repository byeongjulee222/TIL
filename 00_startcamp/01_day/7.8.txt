import requests
from bs4 import BeautifulSoup

html = requests.get('https://finance.naver.com/sise/').text   # 1. 원하는 주소로 요청을 보내 응답을 저장한다.
soup = BeautifulSoup(html, 'html.parser')   # 2. 정보를 조작하기 편하게 바꾸고(정제)
kospi = soup.select_one('#KOSPI_now').text   # 3. 바꾼 정보 중 원하는 정보만 추출
print(kospi)

6) 환율 빼오기
import requests
from bs4 import BeautifulSoup

html = requests.get('https://finance.naver.com/marketindex/')    # 정보 가져옴
soup = BeautifulSoup(html, 'html.parser')    # 정보 가공(타입 변경)
exchange = soup.select_one('#exchangeList > li.on > a.head.usd > div > span.value').text    # 필요한 정보만 빼냄
print(exchange)
